```bash

```

## Usage

```jsx
import React, { Component } from 'react'

import { Reheaded } from 'reheaded'

class Example extends Component {
  /* event hanlders */

  render () {
    return (
      <Reheaded
        onPin={this.onPin}
        onUnpin={this.onUnpin}
        onUnfix={this.onUnfix}
      >
        {({ setRef, height, state }) => (
          <div
            style={{
              height,
            }}
          >
            <Header
              innerRef={setRef}
              state={state}
            >
              Reheaded
            </Header>
          </div>
        )}
      </Reheaded>
    )
  }
}
```
